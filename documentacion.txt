Conexion base de datos
-----------------------
import mysql from "mysql";
import atributos_base_datos from "./atributos_base_datos";


const conexion = mysql.createConnection(atributos_base_datos.parametros);

// Con esto probamos la conexion
conexion.connect(function(error){
      if(error){
          console.log(error);
        
      }else{
          console.log('Conectado a la base de datos');
        
      }
 });

// Evento en caso de error
conexion.on('error', function(e){
    console.log(e);
    
});

export default conexion;
